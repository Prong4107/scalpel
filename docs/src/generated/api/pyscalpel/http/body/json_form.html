 38def json_escape_bytes(data: bytes) -> str:
 39    printable = string.printable.encode()
 40
 41    return "".join(
 42        chr(ch) if ch in printable else "\\u{:04x}".format(ch) for ch in data
 43    )

 46def json_unescape(escaped: str) -> str:
 47    def decode_match(match):
 48        return chr(int(match.group(1), 16))
 49
 50    return re.sub(r"\\u([0-9a-fA-F]{4})", decode_match, escaped)

 53def json_unescape_bytes(escaped: str) -> bytes:
 54    return json_unescape(escaped).encode("latin-1")

 68def transform_tuple_to_dict(tup):
 69    """Transforms duplicates keys to list
 70
 71    E.g:
 72    (("key_duplicate", 1),("key_duplicate", 2),("key_duplicate", 3),
 73    ("key_duplicate", 4),("key_uniq": "val") ,
 74    ("key_duplicate", 5),("key_duplicate", 6))
 75    ->
 76    {"key_duplicate": [1,2,3,4,5], "key_uniq": "val"}
 77
 78
 79    Args:
 80        tup (_type_): _description_
 81
 82    Returns:
 83        _type_: _description_
 84    """
 85    result_dict = {}
 86    for pair in tup:
 87        key, value = pair
 88        converted_key: bytes | str
 89        match key:
 90            case bytes():
 91                converted_key = key.removesuffix(b"[]")
 92            case str():
 93                converted_key = key.removesuffix("[]")
 94            case _:
 95                converted_key = key
 96
 97        if converted_key in result_dict:
 98            if isinstance(result_dict[converted_key], list):
 99                result_dict[converted_key].append(value)
100            else:
101                result_dict[converted_key] = [result_dict[converted_key], value]
102        else:
103            result_dict[converted_key] = value
104    return result_dict

107def json_encode_exported_form(
108    exported: TupleExportedForm,
109) -> tuple[tuple[str, str], ...]:
110    """Unicode escape (\\uXXXX) non printable bytes
111
112    Args:
113        exported (TupleExportedForm): The exported form tuple
114
115    Returns:
116        tuple[tuple[str, str], ...]: exported with every values as escaped strings
117    """
118    return tuple(
119        (
120            json_escape_bytes(key),
121            json_escape_bytes(val or b""),
122        )
123        for key, val in exported
124    )

127def encode_JSON_form(
128    d: dict[JSON_KEY_TYPES, JSON_VALUE_TYPES]
129) -> dict[JSON_KEY_TYPES, JSON_VALUE_TYPES]:
130    new_dict = {}
131    for k, v in d.items():
132        new_key = json_unescape(k) if isinstance(k, str) else k
133        if isinstance(v, dict):
134            new_value = encode_JSON_form(v)
135        elif isinstance(v, str):
136            new_value = json_unescape(v)
137        else:
138            new_value = v
139        new_dict[new_key] = new_value
140    return new_dict

143class JSONFormSerializer(FormSerializer):
144    def serialize(
145        self, deserialized_body: Mapping[JSON_KEY_TYPES, JSON_VALUE_TYPES], req=...
146    ) -> bytes:
147        return json.dumps(deserialized_body).encode("utf-8")
148
149    def deserialize(self, body: bytes, req=...) -> JSONForm | None:
150        try:
151            parsed = json.loads(body)
152        except json.JSONDecodeError:
153            return None
154
155        return JSONForm(parsed) if isinstance(parsed, dict) else None
156
157    def get_empty_form(self, req=...) -> JSONForm:
158        return JSONForm()
159
160    def deserialized_type(self) -> type[JSONForm]:
161        return JSONForm
162
163    def export_form(self, source: JSONForm) -> TupleExportedForm:
164        unescaped_source = encode_JSON_form(source)
165        # Transform the dict to a php style query string
166        serialized_to_qs = qs.build_qs(unescaped_source)
167
168        # Parse the query string
169        qs_parser = URLEncodedFormSerializer()
170        parsed_qs = qs_parser.deserialize(always_bytes(serialized_to_qs))
171
172        # Export the parsed query string to the tuple format
173        tupled_form = qs_parser.export_form(parsed_qs)
174        return tupled_form
175
176    def import_form(self, exported: ExportedForm, req=...) -> JSONForm:
177        # Parse array keys like "key1[key2][key3]" and place value to the correct path
178        # e.g: ("key1[key2][key3]", "nested_value") -> {"key1": {"key2" : {"key3" : "nested_value"}}}
179        dict_form = qs.qs_parse_pairs(list(json_encode_exported_form(exported)))
180        json_form = JSONForm(dict_form.items())
181        return json_form

Supports the following objects and types by default:

+-------------------+---------------+ | Python | JSON | +===================+===============+ | dict | object | +-------------------+---------------+ | list, tuple | array | +-------------------+---------------+ | str | string | +-------------------+---------------+ | int, float | number | +-------------------+---------------+ | True | true | +-------------------+---------------+ | False | false | +-------------------+---------------+ | None | null | +-------------------+---------------+

To extend this to recognize other objects, subclass and implement a .default() method with another method that returns a serializable object for o if possible, otherwise it should call the superclass implementation (to raise TypeError).
