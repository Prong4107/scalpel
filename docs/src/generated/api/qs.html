116def build_qs(query: Mapping) -> str:
117    def dict_generator(indict, pre=None):
118        pre = pre[:] if pre else []
119        if isinstance(indict, dict):
120            for key, value in indict.items():
121                if isinstance(value, dict):
122                    for d in dict_generator(value, pre + [key]):
123                        yield d
124                else:
125                    yield pre + [key, value]
126        else:
127            yield indict
128
129    paths = [i for i in dict_generator(query)]
130    qs = []
131
132    for path in paths:
133        names = path[:-1]
134        value = path[-1]
135        s: list[str] = []
136        for i, n in enumerate(names):
137            n = f"[{n}]" if i > 0 else str(n)
138            s.append(n)
139
140        match value:
141            case list() | tuple():
142                for v in value:
143                    multi = s[:]
144                    if not s[-1].endswith("[]"):
145                        multi.append("[]")
146                    multi.append("=")
147                    # URLEncode value
148                    multi.append(quote_plus(str(v)))
149                    qs.append("".join(multi))
150            case _:
151                s.append("=")
152                # URLEncode value
153                s.append(quote_plus(str(value)))
154                qs.append("".join(s))
155
156    return "&".join(qs)
